4) Properties (getters and setters)
    > example
        struct Monster
        {
            public string Name { get; set; } // set variable with properties
            public bool DoubleDamage { get; set; }

            public int Damage
            {
                get // custom getters (override)
                {
                    if (DoubleDamage)
                    {
                        return _damage * 2;
                    }
                    return _damage;
                }
                set
                {
                    _damage = value;
                }
            }
        }

            private int _damage; // backing value (ask about it?)
